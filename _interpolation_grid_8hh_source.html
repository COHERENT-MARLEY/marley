212  inline bool find_bin_limits(FirstNumericType x,
213  GridConstIterator& lower_point, GridConstIterator& upper_point) const
214  {
215  // Check to make sure that the grid contains at least two ordered pairs
216  check_grid();
217 
218  // Find the first point on the grid that is not less than x
219  bool extrapolate = false;
220  GridConstIterator begin = ordered_pairs_.begin();
221  GridConstIterator end = ordered_pairs_.end();
222  GridConstIterator not_less_point = lower_bound(begin, end, x);
223 
224  // Check whether the requested grid point is within the grid limits
225  if (not_less_point == begin) {
226  lower_point = begin;
227  upper_point = begin + 1;
228  // First element of xs > x
229  if (begin->first != x) extrapolate = true;
230  }
231  else if (not_less_point == end) {
232  // last element of xs < x (extrapolate on the right)
233  extrapolate = true;
234  lower_point = end - 2;
235  upper_point = end - 1;
236  }
237  else {
238  // x is within the grid limits
239  lower_point = not_less_point - 1;
240  upper_point = not_less_point;
241  }
242 
243  return extrapolate;
244  }

248  SecondNumericType InterpolationGrid<FirstNumericType,
249  SecondNumericType>::interpolate(FirstNumericType x) const
250  {
251  // Find grid points just below [(x1, y1)] and just above [(x2, y2)]
252  InterpolationGrid::GridConstIterator lower_point, upper_point;
253  // Includes a call to check_grid()
254  bool extrapolate = find_bin_limits(x, lower_point, upper_point);
255 
256  // If the requested x value is outside of the grid, use the correct method
257  // for dealing with this situation based on the value of
258  if (extrapolate) {
259  if (extrapolation_method_ == ExtrapolationMethod::Zero)
260  return static_cast<SecondNumericType>(0.);
261  else if (extrapolation_method_ == ExtrapolationMethod::Endpoint) {
262  if (lower_point->first > x) return lower_point->second;
263  else return upper_point->second;
264  }
265  else if (extrapolation_method_ == ExtrapolationMethod::Throw) {
266  throw marley::Error(std::string("x = ") + std::to_string(x)
267  + " lies outside of the current interpolation grid object"
268  + " (which extends from x_min = "
269  + std::to_string(ordered_pairs_.front().first)
270  + " and x_max = " + std::to_string(ordered_pairs_.back().first)
271  + ") and extrapolation is disabled.");
272  }
273  }
274 
275  // Either the requested point falls within the grid or
276  // extrapolation_method == ExtrapolationMethod::Continue. If the "continue"
277  // method for extrapolation is selected, then we continue to use the same
278  // interpolation technique formula outside of the grid (for
279  // InterpolationMethod::LinearLinear, this is linear-linear extrapolation).
280 
281  // If the constant interpolation method is selected, then use the *lower
282  // bound* of each bin as the interpolated value. If we're extrapolating,
283  // on the right, use the upper bound.
284  if (interpolation_method_ == InterpolationMethod::Constant) {
285  if (!extrapolate) return lower_point->second;
286  else if (lower_point->first > x) return lower_point->second;
287  else return upper_point->second;
288  }
289 
290  // We'll use an interpolation formula for all other methods, so
291  // get the grid point x and y values for later use.
292  FirstNumericType x1 = lower_point->first;
293  FirstNumericType x2 = upper_point->first;
294  SecondNumericType y1 = lower_point->second;
295  SecondNumericType y2 = upper_point->second;
296 
297  bool log_x = false, log_y = false;
298  FirstNumericType x_to_use = x;
299  if (interpolation_method_ == InterpolationMethod::LinearLog)
300  log_x = true;
301  else if (interpolation_method_ == InterpolationMethod::LogLinear)
302  log_y = true;
303  else if (interpolation_method_ == InterpolationMethod::LogLog) {
304  log_x = true;
305  log_y = true;
306  }
307  if (log_x) {
308  x1 = std::log(x1);
309  x2 = std::log(x2);
310  x_to_use = std::log(x);
311  }
312  if (log_y) {
313  y1 = std::log(y1);
314  y2 = std::log(y2);
315  }
316 
317  FirstNumericType y_interp = y1 + ((y2 - y1)/(x2 - x1))*(x_to_use - x1);
318  if (log_y) y_interp = std::exp(y_interp);
319  return y_interp;
320  }
