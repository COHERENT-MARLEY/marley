221  inline bool find_bin_limits(FirstNumericType x,
222  GridConstIterator& lower_point, GridConstIterator& upper_point) const
223  {
224  // Check to make sure that the grid contains at least two ordered pairs
225  check_grid();
226 
227  // Find the first point on the grid that is not less than x
228  bool extrapolate = false;
229  GridConstIterator begin = ordered_pairs_.begin();
230  GridConstIterator end = ordered_pairs_.end();
231  GridConstIterator not_less_point = lower_bound(begin, end, x);
232 
233  // Check whether the requested grid point is within the grid limits
234  if (not_less_point == begin) {
235  lower_point = begin;
236  upper_point = begin + 1;
237  // First element of xs > x
238  if (begin->first != x) extrapolate = true;
239  }
240  else if (not_less_point == end) {
241  // last element of xs < x (extrapolate on the right)
242  extrapolate = true;
243  lower_point = end - 2;
244  upper_point = end - 1;
245  }
246  else {
247  // x is within the grid limits
248  lower_point = not_less_point - 1;
249  upper_point = not_less_point;
250  }
251 
252  return extrapolate;
253  }

256  template <typename FirstNumericType, typename SecondNumericType>
257  SecondNumericType InterpolationGrid<FirstNumericType,
258  SecondNumericType>::interpolate(FirstNumericType x) const
259  {
260  // Find grid points just below [(x1, y1)] and just above [(x2, y2)]
261  InterpolationGrid::GridConstIterator lower_point, upper_point;
262  // Includes a call to check_grid()
263  bool extrapolate = find_bin_limits(x, lower_point, upper_point);
264 
265  // If the requested x value is outside of the grid, use the correct method
266  // for dealing with this situation based on the value of
267  if (extrapolate) {
268  if (extrapolation_method_ == ExtrapolationMethod::Zero)
269  return static_cast<SecondNumericType>(0.);
270  else if (extrapolation_method_ == ExtrapolationMethod::Endpoint) {
271  if (lower_point->first > x) return lower_point->second;
272  else return upper_point->second;
273  }
274  else if (extrapolation_method_ == ExtrapolationMethod::Throw) {
275  throw marley::Error(std::string("x = ") + std::to_string(x)
276  + " lies outside of the current interpolation grid object"
277  + " (which extends from x_min = "
278  + std::to_string(ordered_pairs_.front().first)
279  + " and x_max = " + std::to_string(ordered_pairs_.back().first)
280  + ") and extrapolation is disabled.");
281  }
282  }
283 
284  // Either the requested point falls within the grid or
285  // extrapolation_method == ExtrapolationMethod::Continue. If the "continue"
286  // method for extrapolation is selected, then we continue to use the same
287  // interpolation technique formula outside of the grid (for
288  // InterpolationMethod::LinearLinear, this is linear-linear extrapolation).
289 
290  // If the constant interpolation method is selected, then use the *lower
291  // bound* of each bin as the interpolated value. If we're extrapolating,
292  // on the right, use the upper bound.
293  if (interpolation_method_ == InterpolationMethod::Constant) {
294  if (!extrapolate) return lower_point->second;
295  else if (lower_point->first > x) return lower_point->second;
296  else return upper_point->second;
297  }
298 
299  // We'll use an interpolation formula for all other methods, so
300  // get the grid point x and y values for later use.
301  FirstNumericType x1 = lower_point->first;
302  FirstNumericType x2 = upper_point->first;
303  SecondNumericType y1 = lower_point->second;
304  SecondNumericType y2 = upper_point->second;
305 
306  bool log_x = false, log_y = false;
307  FirstNumericType x_to_use = x;
308  if (interpolation_method_ == InterpolationMethod::LinearLog)
309  log_x = true;
310  else if (interpolation_method_ == InterpolationMethod::LogLinear)
311  log_y = true;
312  else if (interpolation_method_ == InterpolationMethod::LogLog) {
313  log_x = true;
314  log_y = true;
315  }
316  if (log_x) {
317  x1 = std::log(x1);
318  x2 = std::log(x2);
319  x_to_use = std::log(x);
320  }
321  if (log_y) {
322  y1 = std::log(y1);
323  y2 = std::log(y2);
324  }
325 
326  FirstNumericType y_interp = y1 + ((y2 - y1)/(x2 - x1))*(x_to_use - x1);
327  if (log_y) y_interp = std::exp(y_interp);
328  return y_interp;
329  }
